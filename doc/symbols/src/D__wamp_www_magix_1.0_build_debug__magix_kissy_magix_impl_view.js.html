1 /**
  2  * @fileOverview view中根据底层类库需要重写的方法
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 KISSY.add("magix/impl/view",function(S,io,Sizzle,Magix){
  7 	var IView=function(){
  8 
  9 	};
 10 	var StaticWhiteList={
 11 		idIt:1,
 12 		wrapAsyncUpdate:1,
 13 		registerAsyncUpdateName:1,
 14 		extend:1
 15 	};
 16 	var ex=function(props,ctor){
 17 		var me=this;
 18 		var fn=function(){
 19 			fn.superclass.constructor.apply(this,arguments);
 20 			if(ctor){
 21 				Magix.safeExec(ctor,[],this);
 22 			}
 23 		}
 24 		for(var p in me){
 25 			if(Magix.hasProp(StaticWhiteList,p)){
 26 				fn[p]=me[p];
 27 			}
 28 		}
 29 		return S.extend(fn,me,props);
 30 	};
 31 
 32 	IView.extend=ex;
 33 
 34 	Magix.mix(IView.prototype,{
 35 		getTmplByXHR:function(path,fn){
 36 			io({
 37 				url:path,
 38 				dataType:'html',
 39 				success:function(tmpl){
 40 					fn(tmpl);
 41 				},
 42 				error:function(e,msg){
 43 					fn(msg);
 44 				}
 45 			});
 46 		},
 47 		delegateUnbubble:function(node,event){
 48 			var me=this;
 49 			if(!me.$cacheEvents)me.$cacheEvents={};
 50 			node=S.one(node);
 51 			node.delegate(event,'*[mx'+event+']',me.$cacheEvents[event]=function(e){
 52 				me.processEvent(e);
 53 			});
 54 		},
 55 		undelegateUnbubble:function(node,event){
 56 			var me=this;
 57 			var cache=me.$cacheEvents;
 58 			if(cache){
 59 				node=S.one(node);
 60 				//
 61 				node.undelegate(event,'*[mx'+event+']',cache[event]);
 62 				delete cache[event];
 63 			}
 64 		}
 65 	});
 66 
 67 	return IView;
 68 },{
 69 	requires:["ajax","sizzle","magix/magix"]
 70 });
