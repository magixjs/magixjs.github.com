1 /**
  2  * @fileOverview 对magix/view的扩展
  3  * @version 1.0

  5  */
  6 KISSY.add('mxext/view',function(S,View,Router){
  7 	var WIN=window;
  8 	/*
  9 		queryEvents:{
 10 			click:{
 11 				'#id':function(){
 12 					
 13 				},
 14 				'.title':function(){//  S.one('.title').click(); S.one().delegate(); 
 15 					
 16 				}
 17 			},
 18 			mouseover:{
 19 				'#id':function(e){
 20 					
 21 				}
 22 			}
 23 		}
 24 	 */
 25 	/**
 26 	 * @name MxView
 27 	 * @namespace
 28 	 * @requires View
 29 	 * @augments View
 30 	 */
 31 	return View.extend({
 32 		/**
 33 		 * @lends MxView#
 34 		 */
 35 		/**
 36 		 * 根据选择器来注册事件
 37 		 * @type {Object}
 38 		 * @example
 39 		 * queryEvents:{
 40 		 * 		click:{
 41 		 * 			'#name':function(e){
 42 		 * 				
 43 		 * 			},
 44 		 * 			'#name .label':function(e){
 45 		 * 				
 46 		 * 			}
 47 		 * 		}
 48 		 * }
 49 		 */
 50 		queryEvents:{},
 51 		/**
 52 		 * 调用magix/router的navigate方法
 53 		 * @param {Object|String} params 参数字符串或参数对象
 54 		 */
 55 		navigate:function(params){
 56 			Router.navigate(params);
 57 		},
 58 		/**
 59 		 * 根据选择器添加事件
 60 		 */
 61 		attachQueryEvents:function(){
 62 			var me=this;
 63 			me.$queryEventsCache={};
 64 			var queryEvents=me.queryEvents;
 65 			if(queryEvents){
 66 				for(var p in queryEvents){
 67 					var evts=queryEvents[p];
 68 					for(var q in evts){
 69 						
 70 						S.all('#'+me.id+' '+q).on(p,me.$queryEventsCache[p+'_'+q]=(function(fn){
 71 							return function(e){
 72 								var targetId=View.idIt(e.target);
 73 								var currentId=View.idIt(e.currentTarget);
 74 								Magix.safeExec(fn,{
 75 									view:me,
 76 									targetId:targetId,
 77 									currentId:currentId,
 78 									queryEvents:queryEvents,
 79 									domEvent:e
 80 								},queryEvents);
 81 							}
 82 						}(evts[q])));
 83 					}
 84 				}
 85 			}
 86 			
 87 		},
 88 		/**
 89 		 * 清除根据选择器添加的事件
 90 		 */
 91 		detachQueryEvents:function(){
 92 			var me=this;
 93 			me.$queryEventsCache={};
 94 			var queryEvents=me.queryEvents;
 95 			if(queryEvents){
 96 				for(var p in queryEvents){
 97 					var evts=queryEvents[p];
 98 					for(var q in evts){
 99 						S.all('#'+me.id+' '+q).detach(p,me.$queryEventsCache[p+'_'+q]);
100 					}
101 				}
102 			}
103 		}
104 	},function(){
105 		var me=this;
106 		me.bind('created',function(){
107 			me.attachQueryEvents();
108 			me.bind('refresh',function(){
109 				me.detachQueryEvents();
110 			});
111 			me.bind('rendered',function(){
112 				me.attachQueryEvents();
113 			});
114 		});
115 	});
116 },{
117 	requires:["magix/view","magix/router"]
118 });
