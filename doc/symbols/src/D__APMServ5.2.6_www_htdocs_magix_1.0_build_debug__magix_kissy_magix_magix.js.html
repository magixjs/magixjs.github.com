  8 /**
  9  * 检测某个对象是否拥有某个属性
 10  * @param  {Object}  owner 检测对象
 11  * @param  {String}  prop  属性
 12  * @return {Boolean} 是否拥有prop属性 
 13  */
 14 var has=function(owner,prop){
 15 	if(!owner)return false;//false 0 null '' undefined
 16 	return Object.prototype.hasOwnProperty.call(owner,prop);
 17 };

 19 /**
 20  * 混合对象的属性
 21  * @param  {Object} aim    要mix的目标对象
 22  * @param  {Object} src    mix的来源对象
 23  * @param  {Object} ignore 在复制时，忽略的值
 24  * @return {Object}
 25  */
 26 var mix=function(aim,src,ignore){
 27 	for(var p in src){
 28 		if(has(src,p)&&(!ignore||!has(ignore,p))){
 29 			aim[p]=src[p]
 30 		}
 31 	}
 32 	return aim;
 33 };

 35 /**
 36  * 以try cache方式执行方法，忽略掉任何异常
 37  * @param  {Array} fns     函数数组
 38  * @param  {Array} args    参数数组
 39  * @param  {Object} context 在待执行的方法内部，this的指向
 40  * @return {Object} 返回执行的最后一个方法的返回值
 41  */
 42 var safeExec=function(fns,args,context,i,r,e){
 43 	if(!Magix.isArray(fns)){
 44 		fns=[fns];
 45 	}
 46 	if(!args||(!Magix.isArray(args)&&!args.callee)){
 47 		args=[args];
 48 	}
 49 	for(i=0;i<fns.length;i++){
 50 		try{
 51 			e=fns[i];
 52 			r=Magix.isFunction(e)&&e.apply(context,args);
 53 		}catch(x){
 54 			
 55 		}
 56 	}
 57 	return r
 58 };

232 	 * @param {Boolean} cfg.useHistoryState 是否使用history state,当为true，并且浏览器支持的情况下会用history.pushState修改url，您应该确保服务器能给予支持。如果useHistoryState为false将使用hash修改url

284 	/**
285 	 * 获取对象的keys
286 	 * @param  {Object} obj 要获取key的对象
287 	 * @return {Array}
288 	 */
289 	objectKeys:function(obj){
290 		if(Object.keys){
291 			return Object.keys(obj);
292 		}else{
293 			var keys=[];
294 			for(var p in obj){
295 				if(has(obj,p)){
296 					keys.push(p);
297 				}
298 			}
299 			return keys;
300 		}

302 	/**
303 	 * 比较2个参数对象值是否一样，浅比较
304 	 * @param  {Object} p1 给定的第一个对象
305 	 * @param  {Object} p2 给定的第二个对象
306 	 * @return {Boolean} 2个对象的内容是否一样
307 	 */
308 	compareObject:function(p1,p2){
309 		var me=this;
310 		if(me.isObject(p1)&&me.isObject(p2)){
311 			var keys=[];
312 			keys=keys.concat(me.objectKeys(p1));
313 			keys=keys.concat(me.objectKeys(p2));
314 			for(var i=0;i<keys.length;i++){
315 				var key=keys[i];
316 				if(p1[key]!=p2[key]){
317 					return false;
318 				}
319 			}
320 			return true;
321 		}else{
322 			return p1==p2;
323 		}
324 	},

327 	 * @param {String|Object} key 获取或设置Magix.locals时的key 或者 设置Magix.locals的对象
328 	 * @param {[type]} [val] 设置的对象
329 	 * @return {Object|Undefined}
330 	 * @example
331 	 * Magix.local({//以对象的形式存值
332 	 * 		userId:'58782'
333 	 * });

342 	 */

344 		var args=arguments;

346 		var locals=me.locals;
347 		if(args.length==0){
348 			return locals;
349 		}else if(args.length==1){
350 			if(me.isObject(key)){
351 				for(var p in key){
352 					if(has(key,p)){
353 						locals[p]=key[p];
354 					}
355 				}
356 			}else{
357 				return locals[key];
358 			}
359 		}else{
360 			locals[key]=val;
361 		}
362 	}
363 };
