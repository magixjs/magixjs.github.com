1 /**
  2  * @fileOverview VOM
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 KISSY.add("magix/vom",function(S,Vframe,Magix){
  7 	var D=document;
  8 
  9 /**
 10  * VOM对象
 11  * @name VOM
 12  * @namespace
 13  */
 14 var VOM={
 15 	/**
 16 	 * @lends VOM
 17 	 */
 18 	/**
 19 	 * 根vframe对象
 20 	 * @type {Vframe}
 21 	 */
 22 	rootVframe:null,
 23 	/**
 24 	 * 注册的vframes集合
 25 	 * @type {Object}
 26 	 */
 27 	vframes:{},
 28 	/**
 29 	 * 根vframe的id
 30 	 * @default magix_vf_root
 31 	 * @type {String}
 32 	 */
 33 	rootVframeId:'magix_vf_root',
 34 	/**
 35 	 * 注册vframe对象
 36 	 * @param {Vframe} vf Vframe对象
 37 	 */
 38 	registerVframe:function(vf){
 39 		var me=this;
 40 		me.vframes[vf.id]=vf;
 41 	},
 42 	/**
 43 	 * 根据vframe的id获取vframe对象
 44 	 * @param {String} id vframe的id
 45 	 * @return {Vframe} vframe对象
 46 	 */
 47 	getVframe:function(id){
 48 		return this.vframes[id];
 49 	},
 50 	/**
 51 	 * 删除已注册的vframe对象
 52 	 * @param {Vframe|String} vf vframe对象或对象的id
 53 	 */
 54 	unregisterVframe:function(vf){
 55 		var id=Magix.isString(vf)?vf:vf.id;
 56 		delete this.vframes[id];
 57 	},
 58 	/**
 59 	 * 构建根vframe对象
 60 	 */
 61 	buildRootVframe:function(){
 62 		var me=this;
 63 		if(!me.rootVframe){
 64 			var rootVframeNode=D.getElementById(me.rootVframeId);
 65 			if(!rootVframeNode){//当发现不存在的节点时，由Vframe对象负责创建
 66 				Vframe.createVframeNode(me.rootVframeId,D.body.firstChild);
 67 			}
 68 			me.rootVframe=Vframe.createVframe(me.rootVframeId,{owner:me});
 69 			me.registerVframe(me.rootVframe);
 70 		}
 71 	},
 72 	/**
 73 	 * 重新渲染根vframe
 74 	 * @param {Object} location 当前地址栏解析出来的对象
 75 	 */
 76 	remountRootVframe:function(location){
 77 		//
 78 		var me=this;
 79 		me.$location=location;
 80 		me.buildRootVframe();
 81 		//
 82 		me.rootVframe.mountView(location.viewPath);
 83 	},
 84 	/**
 85 	 * 向vframe通知地址栏发生变化
 86 	 * @param {Object} location 当前地址栏解析出来的对象
 87 	 * @param {Object} changed  前地址栏解析出来的对象与上次解析出来的对象，之间改变的部分
 88 	 */
 89 	notifyLocationChange:function(location,changed){
 90 		var me=this;
 91 		me.$location=location;
 92 		if(me.rootVframe){
 93 			me.rootVframe.notifyLocationChange(location,changed);
 94 		}
 95 	},
 96 	/**
 97 	 * 当VOM处于闲置状态时回调传入的fn，当VOM忙碌时则把fn添加到等待队列
 98 	 * @param {Function} fn 闲置时的回调函数
 99 	 * @param {Array} [args] 参数
100 	 * @param {Object} [context] fn内this指向
101 	 */
102 	idle:function(fn,args,context){
103 		var me=this;
104 		//
105 		if(me.$busy){
106 			if(!me.$idleList)me.$idleList=[];
107 			me.$idleList.push({f:fn,a:args,c:context});
108 		}else{
109 			Magix.safeExec(fn,args,context);
110 		}
111 	},
112 	/**
113 	 * 挂起VOM，等待外部的操作完成
114 	 */
115 	suspend:function(){
116 		var me=this;
117 		me.$busy=true;
118 	},
119 	/**
120 	 * 恢复挂起的VOM
121 	 */
122 	resume:function(){
123 		var me=this;
124 		delete me.$busy;
125 		me.execWaitQueue();
126 	},
127 	/**
128 	 * 运行VOM在挂起状态下堆积的等待函数
129 	 */
130 	execWaitQueue:function(){
131 		var me=this,
132 			list=me.$idleList;
133 		if(list&&list.length){
134 			var j=list.length;
135 			var i=0;
136 			while(i<j){
137 				i++;
138 				var o=list.shift();
139 				me.idle(o.f,o.a,o.c);
140 			}
141 		}
142 	}
143 }
144 	return VOM;
145 },{
146 	requires:["magix/vframe","magix/magix"]
147 });
