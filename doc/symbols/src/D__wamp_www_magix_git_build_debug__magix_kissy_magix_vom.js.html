1 /**
  2  * @fileOverview VOM
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 KISSY.add("magix/vom",function(S,Vframe,Magix,Event){
  7     var D=document;
  8 var safeExec=Magix.safeExec;
  9 /**
 10  * VOM对象
 11  * @name VOM
 12  * @namespace
 13  */
 14 var VOM=Magix.mix({
 15     /**
 16      * @lends VOM
 17      */
 18     iC:0,
 19     iQ:[],
 20     /**
 21      * 根vframe对象
 22      * @type {Vframe}
 23      */
 24     rootVframe:null,
 25     /**
 26      * 注册的vframes集合
 27      * @type {Object}
 28      */
 29     vframes:{},
 30     /**
 31      * 根vframe的id
 32      * @default magix_vf_root
 33      * @type {String}
 34      */
 35     rootVframeId:'magix_vf_root',
 36     /**
 37      * 注册vframe对象
 38      * @param {Vframe} vf Vframe对象
 39      */
 40     registerVframe:function(vf){
 41         var me=this;
 42         me.vframes[vf.id]=vf;
 43         me.trigger(vf.id,{vframe:vf},true);
 44     },
 45     /**
 46      * 根据vframe的id获取vframe对象
 47      * @param {String} id vframe的id
 48      * @return {Vframe} vframe对象
 49      */
 50     getVframe:function(id){
 51         return this.vframes[id];
 52     },
 53     /**
 54      * 删除已注册的vframe对象
 55      * @param {Vframe|String} vf vframe对象或对象的id
 56      */
 57     unregisterVframe:function(vf){
 58         var id=Magix.isString(vf)?vf:vf.id;
 59         delete this.vframes[id];
 60         this.unbind(id);
 61     },
 62     /**
 63      * 构建根vframe对象
 64      */
 65     buildRootVframe:function(){
 66         var me=this;
 67         if(!me.rootVframe){
 68             var rootVframeNode=D.getElementById(me.rootVframeId);
 69             if(!rootVframeNode){//当发现不存在的节点时，由Vframe对象负责创建
 70                 Vframe.createVframeNode(me.rootVframeId,D.body.firstChild);
 71             }
 72             me.rootVframe=Vframe.createVframe(me.rootVframeId,{owner:me});
 73             me.registerVframe(me.rootVframe);
 74         }
 75     },
 76     /**
 77      * 重新渲染根vframe
 78      * @param {Object} e Router.locationChanged事件对象
 79      * @param {Object} e.location window.location.href解析出来的对象
 80      * @param {Object} e.changed 包含有哪些变化的对象
 81      */
 82     remountRootVframe:function(e){
 83         //
 84         var me=this;
 85         me.$location=e.location;
 86         me.buildRootVframe();
 87         //
 88         me.rootVframe.mountView(e.location.viewPath);
 89     },
 90     /**
 91      * 向vframe通知地址栏发生变化
 92      * @param {Object} e 事件对象
 93      * @param {Object} e.location window.location.href解析出来的对象
 94      * @param {Object} e.changed 包含有哪些变化的对象
 95      */
 96     notifyLocationChange:function(e){
 97         var me=this;
 98         me.$location=e.location;
 99         if(me.rootVframe){
100             me.rootVframe.notifyLocationChange(e);
101         }
102     },
103     /**
104      * 当VOM处于闲置状态时回调传入的fn，当VOM忙碌时则把fn添加到等待队列
105      * @param {Function} fn 闲置时的回调函数
106      * @param {Array} [args] 参数
107      * @param {Object} [context] fn内this指向
108      */
109     idle:function(fn,args,context){
110         var me=this;
111         //
112         if(me.iC){
113             me.iQ.push([fn,args,context]);
114         }else{
115             Magix.safeExec(fn,args,context);
116         }
117     },
118     /**
119      * 挂起VOM，等待外部的操作完成
120      */
121     suspend:function(){
122         var me=this;
123         me.iC++;
124     },
125     /**
126      * 恢复挂起的VOM
127      */
128     resume:function(){
129         var me=this;
130         if(me.iC>0){
131             me.iC--;
132         }
133         if(!me.iC){
134             var list=me.iQ;
135             if(list.length){
136                 var tasks=[].slice.call(list);
137                 me.iQ=[];
138                 while(tasks.length){
139                     var o=tasks.shift();
140                     me.idle.apply(me,o);
141                 }
142             }
143         }
144     },
145     /**
146      * 向某个vframe发送消息
147      * @param {Array|String} aims  目标vframe id数组
148      * @param {Object} args 消息对象
149      */
150     postMessage:function(aims,args){
151         var me=this;
152         if(!Magix.isArray(aims)){
153             aims=[aims];
154         }
155         if(!args)args={};
156         for(var i=0;i<aims.length;i++){
157             var vframe=me.getVframe(aims[i]);
158             //
159             if(vframe){
160                 vframe.message(args);
161             }else{
162                 me.bind(aims[i],function(e){
163                     
164                     e.vframe.message(args);
165                 });
166             }
167         }
168     }
169 },Event);
170     return VOM;
171 },{
172     requires:["magix/vframe","magix/magix","magix/event"]
173 });
