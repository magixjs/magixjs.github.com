1 /**
  2  * @fileOverview view转场动画
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 KISSY.add('mxext/vfanim',function(S,Vf,Magix){
  7     /**
  8      * view转场动画实现
  9      * @name VfAnim
 10      * @namespace
 11      * @example
 12      * //当使用此插件时，您应该增加一项effect的配置来定制动画效果
 13      * //如：
 14      * 
 15      * Magix.start({
 16      *  //...其它配置项
 17      *  effect:function(e){
 18      *          var S=KISSY;
 19      *      var offset=S.one(e.oldViewNode).offset();
 20      *      S.one(e.oldViewNode).css({backgroundColor:'#fff'});
 21      *      var distance=offset.top+S.one(e.oldViewNode).height();
 22      *      new S.Anim(e.oldViewNode,{top:-distance,opacity:0.2},1.2,'backIn',e.collectGarbage).run();
 23      *      S.one(e.newViewNode).css({opacity:0});
 24      *      new S.Anim(e.newViewNode,{opacity:1},2).run();
 25      *  }
 26      * });
 27      * 
 28      * //参数说明：
 29      * //e.vframeId {Strig} 在哪个vframe内发生的转场动画
 30      * //e.oldViewNode {HTMLElement} 原来的view DOM节点
 31      * //e.newViewNode {HTMLElement} 新创建的view DOM节点
 32      * //e.action {String} 指示是哪种场景改变：viewChange view发生改变 viewRefresh view刷新
 33      * //e.collectGarbage {Function} 当您在外部结束动画时，调用该方法清理不需要的垃圾DOM节点，请优先考虑该方法进行垃圾收集
 34      *
 35      *
 36      * //关于转场动画说明：
 37      * //转场动画仅适用对于同一个vframe渲染不同view时较安全，因为转场时页面上会同时存在
 38      * //这2个view的html内容，而在view内部DOM操作时选择器通常不会意外的选择到其它节点上
 39      * //
 40      * //对于view的刷新不建议使用动画，因为2个view的html内容一样，DOM选择很容易发生失误
 41      * //如果您需要view在刷新时使用动画，最好在代码中DOM选择器都加上id
 42      * //类似：
 43      *
 44      * render:function(){
 45      *      var inputs=S.all('#'+this.id+' input');//加上id限定
 46      * }
 47      *
 48      * 
 49      */
 50 
 51     var EMPTY='';
 52     var MxConfig=Magix.config();
 53     var D=document;
 54 
 55     var CfgSceneChange=MxConfig.effect;
 56     var CfgSceneChangeIsFn=Magix.isFunction(CfgSceneChange);
 57 
 58     var $=function(id){
 59         return typeof id=='object'?id:D.getElementById(id);
 60     };
 61     return Magix.mix(Vf.prototype,{
 62         useAnimUpdate:function(){
 63             var me=this;
 64             console.log(MxConfig);
 65             return CfgSceneChange;
 66             /*var anim=me.$currentSupportAmin=(Math.random()<0.5)
 67             return anim;*/
 68         },
 69         oldViewDestroy:function(){
 70             var me=this;
 71             var ownerNode=$(me.id);
 72             var oldViewNode=$(me.vId);
 73             var view=me.view;
 74             if(!oldViewNode){
 75                 oldViewNode=D.createElement('div');
 76                 while(ownerNode.firstChild){
 77                     oldViewNode.appendChild(ownerNode.firstChild);
 78                 }
 79                 ownerNode.appendChild(oldViewNode);
 80             }
 81             oldViewNode.id=EMPTY;
 82             var events=view.events;
 83             if(events){
 84                 for(var p in events){
 85                     if(Magix.has(events,p)){
 86                         S.all('*[mx-'+p+']').removeAttr('mx-'+p);
 87                     }
 88                 }
 89             }
 90             //destroy[1,2,4,5,6]
 91             //new[2,4]
 92             //created[4]
 93             if(!me.$tempNodes){
 94                 me.$tempNodes=[];
 95             }
 96             me.$tempNodes.push(oldViewNode);
 97         },
 98         prepareNextView:function(){
 99             var me=this;
100             var ownerNode=$(me.id);
101             var div=D.createElement('div');
102             div.id=me.vId;
103             if(ownerNode._bak){
104                 div.innerHTML=ownerNode._tmpl;
105             }
106             ownerNode.insertBefore(div,ownerNode.firstChild);
107         },
108         newViewCreated:function(isViewChange){
109             var me=this;
110             var oldViewNode=me.$oldViewNode;
111             var newViewNode=$(me.vId);
112             if(!me.$anim){
113                 me.$anim=1;
114             }
115             me.$anim++;
116             if(CfgSceneChangeIsFn){
117                 Magix.safeExec(CfgSceneChange,{
118                     vframeId:me.vfId,
119                     action:isViewChange?'viewChange':'viewRefresh',
120                     oldViewNode:oldViewNode,
121                     newViewNode:newViewNode,
122                     collectGarbage:function(){
123                         me.$anim--;
124                         if(me.$anim==1){
125                             var list=me.$tempNodes;
126                             for(var i=0,e;i<list.length;i++){
127                                 e=list[i];
128                                 e.parentNode.removeChild(e);
129                             }
130                         }
131                     }
132                 },me);
133             }
134         }
135     });
136 },{
137     requires:["magix/vframe","magix/magix","sizzle"]
138 });
