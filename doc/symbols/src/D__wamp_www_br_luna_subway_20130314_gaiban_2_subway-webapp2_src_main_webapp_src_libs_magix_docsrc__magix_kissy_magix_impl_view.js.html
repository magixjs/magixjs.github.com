1 /**
  2  * @fileOverview view中根据底层类库需要重写的方法
  3  * @author 行列
  4  * @version 1.0
  5  */
  6 KISSY.add("magix/impl/view",function(S,io,Magix){
  7     var IView=function(){
  8 
  9     };
 10     var Mods=S.Env.mods;
 11     var StaticWhiteList={
 12         wrapAsyn:1,
 13         extend:1
 14     };
 15 
 16     var processObject=function(props,proto,enterObject){
 17         for(var p in proto){
 18             if(S.isObject(proto[p])){
 19                 if(!Magix.has(props,p))props[p]={};
 20                 processObject(props[p],proto[p],true);
 21             }else if(enterObject){
 22                 props[p]=proto[p];
 23             }
 24         }
 25     };
 26     IView.extend=function(props,ctor){
 27         var me=this;
 28         var BaseView=function(){
 29             BaseView.superclass.constructor.apply(this,arguments);
 30             if(ctor){
 31                 Magix.safeExec(ctor,arguments,this);
 32             }
 33         }
 34         BaseView.extend=IView.extend;
 35         return S.extend(BaseView,me,props);
 36     };
 37 
 38     IView.prepare=function(oView,toProto){
 39         var me=this;
 40         if(!oView.wrapAsyn){
 41             for(var p in me){
 42                 if(Magix.has(StaticWhiteList,p)){
 43                     oView[p]=me[p];
 44                 }
 45             }
 46             var aimObject=oView.prototype;
 47             var start=oView;
 48             var temp;
 49             while(start.superclass){
 50                 temp=start.superclass.constructor;
 51                 processObject(aimObject,temp.prototype);
 52                 start=temp;
 53             }
 54             toProto.home=Mods[toProto.path].packageInfo.getBase();
 55             Magix.mix(aimObject,toProto);
 56         }
 57         oView.wrapAsyn();
 58     };
 59 
 60     Magix.mix(IView.prototype,{
 61         fetchTmpl:function(path,fn,d){
 62             io({
 63                 url:path+(d?'?_='+S.now():''),
 64                 success:fn,
 65                 error:function(e,m){
 66                     fn(m)
 67                 }
 68             });
 69         }
 70     });
 71 
 72     return IView;
 73 },{
 74     requires:["ajax","magix/magix"]
 75 });
