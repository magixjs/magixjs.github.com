1 KISSY.add("mxext/mustache", function(S) {
  2 
  3 	/*
  4 	 mustache.js — Logic-less templates in JavaScript
  5 
  6 	 See http://mustache.github.com/ for more info.
  7 	 */
  8 	 /**
  9 	  * Mustache模板
 10 	  * @name Mustache
 11 	  * @namespace
 12 	  */
 13 	var Mustache = function() {
 14 		/**
 15 		 * @name Renderer
 16 		 * @inner
 17 		 */
 18 		var Renderer = function() {
 19 		};
 20 
 21 		Renderer.prototype = {
 22 			otag : "{{",
 23 			ctag : "}}",
 24 			pragmas : {},
 25 			buffer : [],
 26 			pragmas_implemented : {
 27 				"IMPLICIT-ITERATOR" : true
 28 			},
 29 			context : {},
 30 
 31 			render : function(template, context, partials, in_recursion) {
 32 				// reset buffer & set context
 33 				if(!in_recursion) {
 34 					this.context = context;
 35 					this.buffer = [];
 36 					// TODO: make this non-lazy
 37 				}
 38 
 39 				// fail fast
 40 				if(!this.includes("", template)) {
 41 					if(in_recursion) {
 42 						return template;
 43 					} else {
 44 						this.send(template);
 45 						return;
 46 					}
 47 				}
 48 				template = this.render_pragmas(template);
 49 				var html = this.render_section(template, context, partials);
 50 				if(in_recursion) {
 51 					return this.render_tags(html, context, partials, in_recursion);
 52 				}
 53 
 54 				this.render_tags(html, context, partials, in_recursion);
 55 			},
 56 			/*
 57 			 Sends parsed lines
 58 			 */
 59 			send : function(line) {
 60 				if(line != "") {
 61 					this.buffer.push(line);
 62 				}
 63 			},
 64 			/*
 65 			 Looks for %PRAGMAS
 66 			 */
 67 			render_pragmas : function(template) {
 68 				// no pragmas
 69 				if(!this.includes("%", template)) {
 70 					return template;
 71 				}
 72 
 73 				var that = this;
 74 				var regex = new RegExp(this.otag + "%([\\w-]+) ?([\\w]+=[\\w]+)?" + this.ctag);
 75 				return template.replace(regex, function(match, pragma, options) {
 76 					if(!that.pragmas_implemented[pragma]) {
 77 						throw ( {
 78 							message : "This implementation of mustache doesn't understand the '" + pragma + "' pragma"
 79 						});
 80 					}
 81 					that.pragmas[pragma] = {};
 82 					if(options) {
 83 						var opts = options.split("=");
 84 						that.pragmas[pragma][opts[0]] = opts[1];
 85 					}
 86 					return "";
 87 					// ignore unknown pragmas silently
 88 				});
 89 			},
 90 			/*
 91 			 Tries to find a partial in the curent scope and render it
 92 			 */
 93 			render_partial : function(name, context, partials) {
 94 				name = this.trim(name);
 95 				if(!partials || partials[name] === undefined) {
 96 					throw ( {
 97 						message : "unknown_partial '" + name + "'"
 98 					});
 99 				}
100 				if( typeof (context[name]) != "object") {
101 					return this.render(partials[name], context, partials, true);
102 				}
103 				return this.render(partials[name], context[name], partials, true);
104 			},
105 			/*
106 			 Renders inverted (^) and normal (#) sections
107 			 */
108 			render_section : function(template, context, partials) {
109 				if(!this.includes("#", template) && !this.includes("^", template)) {
110 					return template;
111 				}
112 
113 				var that = this;
114 				// CSW - Added "+?" so it finds the tighest bound, not the widest
115 				var regex = new RegExp(this.otag + "(\\^|\\#)\\s*(.+)\\s*" + this.ctag + "\n*([\\s\\S]+?)" + this.otag + "\\/\\s*\\2\\s*" + this.ctag + "\\s*", "mg");
116 
117 				// for each {{#foo}}{{/foo}} section do...
118 				return template.replace(regex, function(match, type, name, content) {
119 					var value = that.find(name, context);
120 					if(type == "^") {// inverted section
121 						if(!value || that.is_array(value) && value.length === 0) {
122 							// false or empty list, render it
123 							return that.render(content, context, partials, true);
124 						} else {
125 							return "";
126 						}
127 					} else if(type == "#") {// normal section
128 						if(that.is_array(value)) {// Enumerable, Let's loop!
129 							return that.map(value, function(row) {
130 								return that.render(content, that.create_context(row), partials, true);
131 							}).join("");
132 						} else if(that.is_object(value)) {// Object, Use it as subcontext!
133 							return that.render(content, that.create_context(value), partials, true);
134 						} else if( typeof value === "function") {
135 							// higher order section
136 							return value.call(context, content, function(text) {
137 								return that.render(text, context, partials, true);
138 							});
139 						} else if(value) {// boolean section
140 							return that.render(content, context, partials, true);
141 						} else {
142 							return "";
143 						}
144 					}
145 				});
146 			},
147 			/*
148 			 Replace {{foo}} and friends with values from our view
149 			 */
150 			render_tags : function(template, context, partials, in_recursion) {
151 				// tit for tat
152 				var that = this;
153 
154 				var new_regex = function() {
155 					return new RegExp(that.otag + "(=|!|>|\\{|%)?([^\\/#\\^]+?)\\1?" + that.ctag + "+", "g");
156 				};
157 				var regex = new_regex();
158 				var tag_replace_callback = function(match, operator, name) {
159 					switch(operator) {
160 						case "!":
161 							// ignore comments
162 							return "";
163 						case "=":
164 							// set new delimiters, rebuild the replace regexp
165 							that.set_delimiters(name);
166 							regex = new_regex();
167 							return "";
168 						case ">":
169 							// render partial
170 							return that.render_partial(name, context, partials);
171 						case "{":
172 							// the triple mustache is unescaped
173 							return that.find(name, context);
174 						default:
175 							// escape the value
176 							return that.escape(that.find(name, context));
177 					}
178 				};
179 				var lines = template.split("\n");
180 				for(var i = 0; i < lines.length; i++) {
181 					lines[i] = lines[i].replace(regex, tag_replace_callback, this);
182 					if(!in_recursion) {
183 						this.send(lines[i]);
184 					}
185 				}
186 
187 				if(in_recursion) {
188 					return lines.join("\n");
189 				}
190 			},
191 			set_delimiters : function(delimiters) {
192 				var dels = delimiters.split(" ");
193 				this.otag = this.escape_regex(dels[0]);
194 				this.ctag = this.escape_regex(dels[1]);
195 			},
196 			escape_regex : function(text) {
197 				// thank you Simon Willison
198 				if(!arguments.callee.sRE) {
199 					var specials = ['/', '.', '*', '+', '?', '|', '(', ')', '[', ']', '{', '}', '\\'];
200 					arguments.callee.sRE = new RegExp('(\\' + specials.join('|\\') + ')', 'g');
201 				}
202 				return text.replace(arguments.callee.sRE, '\\$1');
203 			},
204 			/*
205 			 find `name` in current `context`. That is find me a value
206 			 from the view object
207 			 */
208 			find : function(name, context) {
209 				name = this.trim(name);
210 
211 				// Checks whether a value is thruthy or false or 0
212 				function is_kinda_truthy(bool) {
213 					return bool === false || bool === 0 || bool;
214 				}
215 
216 				var value;
217 				if(is_kinda_truthy(context[name])) {
218 					value = context[name];
219 				} else if(is_kinda_truthy(this.context[name])) {
220 					value = this.context[name];
221 				}
222 
223 				if( typeof value === "function") {
224 					return value.apply(context);
225 				}
226 				if(value !== undefined) {
227 					return value;
228 				}
229 				// silently ignore unkown variables
230 				return "";
231 			},
232 			// Utility methods
233 
234 			/* includes tag */
235 			includes : function(needle, haystack) {
236 				return haystack.indexOf(this.otag + needle) != -1;
237 			},
238 			/*
239 			 Does away with nasty characters
240 			 */
241 			escape : function(s) {
242 				s = String(s === null ? "" : s);
243 				return s.replace(/&(?!\w+;)|["'<>\\]/g, function(s) {
244 					switch(s) {
245 						case "&":
246 							return "&";
247 						case "\\":
248 							return "\\\\";
249 						case '"':
250 							return '"';
251 						case "'":
252 							return ''';
253 						case "<":
254 							return "<";
255 						case ">":
256 							return ">";
257 						default:
258 							return s;
259 					}
260 				});
261 			},
262 			// by @langalex, support for arrays of strings
263 			create_context : function(_context) {
264 				if(this.is_object(_context)) {
265 					return _context;
266 				} else {
267 					var iterator = ".";
268 					if(this.pragmas["IMPLICIT-ITERATOR"]) {
269 						iterator = this.pragmas["IMPLICIT-ITERATOR"].iterator;
270 					}
271 					var ctx = {};
272 					ctx[iterator] = _context;
273 					return ctx;
274 				}
275 			},
276 			is_object : function(a) {
277 				return a && typeof a == "object";
278 			},
279 			is_array : function(a) {
280 				return Object.prototype.toString.call(a) === '[object Array]';
281 			},
282 			/*
283 			 Gets rid of leading and trailing whitespace
284 			 */
285 			trim : function(s) {
286 				return s.replace(/^\s*|\s*$/g, "");
287 			},
288 			/*
289 			 Why, why, why? Because IE. Cry, cry cry.
290 			 */
291 			map : function(array, fn) {
292 				if( typeof array.map == "function") {
293 					return array.map(fn);
294 				} else {
295 					var r = [];
296 					var l = array.length;
297 					for(var i = 0; i < l; i++) {
298 						r.push(fn(array[i]));
299 					}
300 					return r;
301 				}
302 			}
303 		};
304 
305 		return ( {
306 			/**
307 			 * @lends Mustache
308 			 */
309 			/**
310 			 * 名称
311 			 */
312 			name : "mustache.js",
313 			/**
314 			 * 版本
315 			 */
316 			version : "0.3.1-dev",
317 
318 			/**
319 			 * 把模板根据数据翻译成最终字符串
320 			 * @param {String} template 模板
321 			 * @param {Object} view     数据
322 			 * @return {String}
323 			 */
324 			to_html : function(template, view, partials, send_fun) {
325 				var renderer = new Renderer();
326 				if(send_fun) {
327 					renderer.send = send_fun;
328 				}
329 				
330 				renderer.render(template, view, partials);
331 				if(!send_fun) {
332 					return renderer.buffer.join("\n");
333 				}
334 			}
335 		});
336 	}();
337 	return Mustache;
338 });
