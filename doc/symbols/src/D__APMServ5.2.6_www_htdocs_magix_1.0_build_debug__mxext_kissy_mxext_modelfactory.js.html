1 /**
  2  * @fileOverview model管理工厂，可方便的对Model进行缓存和更新
  3  * @author 行列
  4  * @version 1.0
  5  **/
  6 KISSY.add("mxext/modelfactory",function(S,Magix){
  7 	/**
  8 	 * 工厂类，可方便的对Model进行缓存和更新
  9 	 * @name ModelFactory
 10 	 * @class
 11 	 * @namespace
 12 	 * @param {Model} modelClass Model类
 13 	 */
 14 	var Factory=function(modelClass){
 15 		var me=this;
 16 		me.$modelClass=modelClass;
 17 	};
 18 	var WrapRequest=function(request){
 19 		if(request._wraped_)return request;
 20 		var req=function(ops){
 21 			var bakSucc=ops.success;
 22 			var model=this;
 23 			ops.success=function(){
 24 				
 25 				model.set('updateIdent',false);
 26 				if(model._after){
 27 					Magix.safeExec(model._after,model);
 28 				}
 29 				bakSucc.apply(ops,arguments);
 30 			}
 31 			request.call(model,ops);
 32 		}
 33 		req._wraped_=true;
 34 		return req;
 35 	};
 36 	Magix.mix(Factory,{
 37 		/**
 38 		 * @lends ModelFactory
 39 		 */
 40 		/**
 41 		 * Model类缓存对象
 42 		 * @type {Object}
 43 		 */
 44 		mClsCache:{},
 45 		/**
 46 		 * 创建Model类工厂对象
 47 		 * @param {String} key        标识key
 48 		 * @param {Model} modelClass Model类
 49 		 */
 50 		create:function(key,modelClass){
 51 			var me=this;
 52 			if(!modelClass){
 53 				throw new Error('Factory.create modelClass ungiven');
 54 			}
 55 			var cache=me.mClsCache;
 56 			if(!key)key=S.guid();
 57 			if(!cache[key]){
 58 				cache[key]=new Factory(modelClass);
 59 			}
 60 			return cache[key];
 61 		}
 62 	});
 63 	var FetchFlags={
 64 		ALL:1,
 65 		ANY:2,
 66 		ONE:4
 67 	};
 68 	Magix.mix(Factory.prototype,{
 69 		/**
 70 		 * @lends ModelFactory#
 71 		 */
 72 		/**
 73 		 * 注册APP中用到的model
 74 		 * @param {Object|Array} models 模块描述信息
 75 		 * @param {String} models.type app中model的唯一标识
 76 		 * @param {Object} models.ops 传递的参数信息，如{uri:'test',isJSONP:true,updateIdent:true}
 77 		 * @param {Object} models.gets 发起请求时，默认的get参数对象
 78 		 * @param {Object} models.posts 发起请求时，默认的post参数对象
 79 		 * @param {Function} models.before model在发起请求前的回调
 80 		 * @param {Function} models.after model在发起请求，并且通过Model.sync调用success后的回调
 81 		 * @example
 82 		 * KISSY.add("app/base/mfactory",function(S,MFctory,Model){
 83 				var MF=MFctory.create('test/mf',Model);
 84 				MF.registerModels([
 85 					{
 86 						type:'Home_List',
 87 						ops:{
 88 							uri:'test'
 89 						},
 90 						gets:{
 91 							a:'12'
 92 						},
 93 						before:function(m){
 94 							
 95 						},
 96 						after:function(m){
 97 							
 98 						}
 99 					},
100 					{
101 						type:'Home_List1',
102 						ops:{
103 							uri:'test'
104 						},
105 						before:function(m){
106 							
107 						},
108 						after:function(m){
109 							
110 						}
111 					}
112 				]);
113 				return MF;
114 			},{
115 				requires:["mxext/modelfactory","app/base/model"]
116 			});
117 
118 			//使用
119 
120 			KISSY.use('app/base/mfactory',function(S,MF){
121 				MF.fetchAll([{
122 					type:MF.Home_List,cacheKey:'aaa',gets:{e:'f'},
123 					type:MF.Home_List1,gets:{a:'b'}
124 				}],function(m1,m2){
125 	
126 				},function(msg){
127 	
128 				});
129 			});
130 		 */
131 		registerModels:function(models){
132 			/*
133 				type:'',
134 				ops:{
135 					uri:'',
136 					isJSONP:'',
137 					updateIdent:false
138 				},
139 				gets:'',
140 				posts:'',
141 				before:function(m){
142 	
143 				},
144 				after:function(m){
145 					
146 				}
147 			 */
148 			var me=this;
149 			if(!Magix.isArray(models)){
150 				models=[models];
151 			}
152 			for(var i=0,model;i<models.length;i++){
153 				model=models[i];
154 				if(!model.type){
155 					throw new Error('model must own a type attribute');
156 				}
157 				me[model.type]=model;
158 			}
159 		},
160 		/**
161 		 * 获取models，该用缓存的用缓存，该发起请求的请求
162 		 * @see ModelFactory#registerModels
163 		 * @param {Object|Array} models 获取models时的描述信息，如:{type:F.Home,cacheKey:'key',gets:{a:'12'},posts:{b:2}}
164 		 * @param {Function} succ   成功时的回调
165 		 * @param {Function} fail   失败时的回调
166 		 * @param {Integer} flag   获取哪种类型的models
167 		 * @return {Object} 返回一个带abort方法的对象，用于取消所有请求的model
168 		 */
169 		fetchModels:function(models,succ,fail,flag){
170 			var me=this;
171 			if(!me.$modelsCache)me.$modelsCache={};
172 			if(!me.$modelsCacheKeys)me.$modelsCacheKeys={};
173 			var modelsCache=me.$modelsCache;
174 			var modelsCacheKeys=me.$modelsCacheKeys;
175 
176 			if(!Magix.isArray(models)){
177 				models=[models];
178 			}
179 			var total=models.length;
180 			var current=0;
181 			var failMsg;
182 
183 			var doneArr=[];
184 			var abortArr=[];
185 			var hasOneSuccess;
186 
187 			var doneFn=function(idx,isFail,model,args){
188 				current++;
189 				if(isFail){
190 					failMsg=args||'fetch data error';
191 				}else{
192 					hasOneSuccess=true;
193 					model.set('updateIdent',false);//成功后标识model不需要更新，防止需要缓存的model下次使用时发起请求
194 					doneArr[idx]=model;
195 					if(model._cacheKey){//需要缓存
196 						modelsCache[model._cacheKey]=model;
197 					}
198 					if(model._after){//有after
199 						Magix.safeExec(model._after,model);
200 					}
201 					if(flag==FetchFlags.ONE){//如果是其中一个成功，则每次成功回调一次
202 						succ(model);
203 					}
204 				}
205 				var cacheKey=model._cacheKey;
206 				if(cacheKey&&Magix.hasProp(modelsCacheKeys,cacheKey)){
207 					var fns=modelsCacheKeys[cacheKey];
208 					delete modelsCacheKeys[cacheKey];
209 					//
210 					Magix.safeExec(fns,[isFail,model,args],model);
211 					//
212 				}
213 				if(flag!=FetchFlags.ONE){
214 					//
215 					if(current>=total){
216 						if(flag==FetchFlags.ANY){//任意一个成功
217 							if(hasOneSuccess){
218 								succ.apply(me,doneArr);
219 							}else{
220 								fail(failMsg);
221 							}
222 						}else{//所有的都要成功
223 							if(!failMsg){
224 								succ.apply(me,doneArr);
225 							}else{
226 								fail(failMsg);
227 							}
228 						}
229 					}
230 				}
231 			};
232 			//
233 			var Slice=Array.prototype.slice;
234 			var wrapDone=function(fn,context){
235 				var a = Slice.call(arguments, 2);
236 				return function(){
237 					return fn.apply(context,a.concat(Slice.call(arguments)));
238 				}
239 			};
240 			for(var i=0,model;i<models.length;i++){
241 				model=models[i];
242 
243 				var cacheKey=model.cacheKey;
244 				var modelEntity;
245 
246 				if(cacheKey&&Magix.hasProp(modelsCacheKeys,cacheKey)){
247 					modelsCacheKeys[cacheKey].push(wrapDone(doneFn,me,i));
248 				}else{
249 					modelEntity=me.create(model,true);
250 					var updateIdent=modelEntity.get('updateIdent');//是否需要更新
251 					if(updateIdent){
252 						abortArr.push(modelEntity);
253 						if(cacheKey){
254 							modelsCacheKeys[cacheKey]=[];
255 						}
256 						modelEntity.request({
257 							success:wrapDone(doneFn,modelEntity,i,false,modelEntity),
258 							error:wrapDone(doneFn,modelEntity,i,true,modelEntity)
259 						});
260 					}else{
261 						doneFn(i,false,modelEntity);
262 					}
263 				}
264 			}
265 			return {
266 				abort:function(){
267 					for(var i=0,m;i<abortArr.length;i++){
268 						m=abortArr[i];
269 						var cacheKey=m._cacheKey;
270 						if(cacheKey&&Magix.hasProp(modelsCacheKeys,cacheKey)){
271 							var fns=modelsCacheKeys[cacheKey];
272 							delete modelsCacheKeys[cacheKey];
273 							Magix.safeExec(fns,[true,m,'abort'],m);
274 						}
275 						m.abort();
276 					}
277 				}
278 			}
279 		},
280 		/**
281 		 * 获取models，所有成功才回调succ，任意一个失败最终会回调fail
282 		 * @param {Object|Array} models 获取models时的描述信息，如:{type:F.Home,cacheKey:'key',gets:{a:'12'},posts:{b:2}}
283 		 * @param {Function} succ   成功时的回调
284 		 * @param {Function} fail   失败时的回调
285 		 * @return {Array} 返回一个带abort方法的对象，用于取消所有请求的model
286 		 */
287 		fetchAll:function(models,succ,fail){
288 			/*
289 				[{type:F.ASide,cacheKey:'',gets:{},posts:{}}]
290 			 */
291 			return this.fetchModels(models,succ,fail,FetchFlags.ALL);
292 		},
293 		/**
294 		 * 获取models，其中任意一个成功最终回调succ，全部失败才回调fail
295 		 * @param {Object|Array} models 获取models时的描述信息，如:{type:F.Home,cacheKey:'key',gets:{a:'12'},posts:{b:2}}
296 		 * @param {Function} succ   成功时的回调
297 		 * @param {Function} fail   失败时的回调
298 		 * @return {Array} 返回一个带abort方法的对象，用于取消所有请求的model
299 		 */
300 		fetchAny:function(models,succ,fail){
301 			return this.fetchModels(models,succ,fail,FetchFlags.ANY);
302 		},
303 		/**
304 		 * 获取models，其中任意一个成功均立即回调，回调会被调用多次
305 		 * @param {Object|Array} models 获取models时的描述信息，如:{type:F.Home,cacheKey:'key',gets:{a:'12'},posts:{b:2}}
306 		 * @param {Function} callback   成功时的回调
307 		 * @return {Array} 返回一个带abort方法的对象，用于取消所有请求的model
308 		 */
309 		fetchOne:function(models,callback){
310 			return this.fetchModels(models,callback,Magix.noop,FetchFlags.ONE);
311 		},
312 		/**
313 		 * 尝试获取缓存的model
314 		 * @param {String} cacheKey 缓存时的key
315 		 * @return {Object} null或缓存的model
316 		 */
317 		getIf:function(cacheKey){
318 			var me=this;
319 			var modelsCache=me.$modelsCache;
320 			if(modelsCache&&Magix.hasProp(modelsCache,cacheKey)){
321 				return modelsCache[cacheKey];
322 			}
323 			return null;
324 		},
325 		/**
326 		 * 设置缓存的model需要更新
327 		 * @param {String} cacheKey 缓存时的key
328 		 */
329 		setUpdateIdent:function(cacheKey){
330 			var me=this;
331 			var model=me.getIf(cacheKey);
332 			if(model){
333 				model.set('updateIdent',true);
334 			}
335 		},
336 		/**
337 		 * 创建model对象
338 		 * @param {Object} model            model描述信息
339 		 * @param {Boolean} doNotWrapRequest 是否不对request进行包装，默认会对model的request进行一次包装，以完成后续的状态更新
340 		 * @return {Model} model对象
341 		 */
342 		create:function(model,doNotWrapRequest){
343 			if(!model.type){
344 				throw new Error('model must own a "type" attribute');
345 			}
346 			var me=this;
347 			var metas=model.type;
348 			var cacheKey=model.cacheKey;
349 			var modelEntity;
350 
351 			if(!me.$modelsCache)me.$modelsCache={};
352 			var modelsCache=me.$modelsCache;
353 
354 			if(cacheKey&&Magix.hasProp(modelsCache,cacheKey)){//缓存
355 				
356 				modelEntity=modelsCache[cacheKey];
357 
358 				var updateIdent=modelEntity.get('updateIdent');
359 				if(updateIdent){//当有更新时，从缓存中删除，防止多次获取该缓存对象导致数据错误
360 					delete modelsCache[cacheKey];
361 				}
362 			}else{
363 				
364 				modelEntity=new me.$modelClass(metas.ops);
365 				modelEntity._after=metas.after;
366 				modelEntity._cacheKey=cacheKey;
367 				modelEntity.set('updateIdent',true);
368 			}
369 			var updateIdent=modelEntity.get('updateIdent');//是否需要更新
370 			if(updateIdent){
371 				modelEntity.reset();
372 				
373 				//默认设置的
374 				modelEntity.setParams(metas.gets);
375 				modelEntity.setPostParams(metas.posts);
376 
377 				//临时传递的
378 				modelEntity.setParams(model.gets);
379 				modelEntity.setPostParams(model.posts);
380 				
381 				if(Magix.isFunction(metas.before)){
382 					Magix.safeExec(metas.before,modelEntity,metas);
383 				}
384 			}
385 			if(!doNotWrapRequest){
386 				modelEntity.request=WrapRequest(modelEntity.request);
387 			}
388 			return modelEntity;
389 		}
390 	});
391 	return Factory;
392 },{
393 	requires:["magix/magix"]
394 });
