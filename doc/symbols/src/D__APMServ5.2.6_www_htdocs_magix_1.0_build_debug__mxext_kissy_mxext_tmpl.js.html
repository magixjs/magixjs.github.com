1 /**
  2  * @fileOverview 模板
  3  * @version 1.0
  4  * @author 行列
  5  */
  6 KISSY.add("mxext/tmpl",function(S){
  7 	var fnCaches={},
  8 		tmplCaches={},
  9 		stack='_'+new Date().getTime(),
 10 		notRender=/\s*<script[^>]+type\s*=\s*(['"])\s*text\/ignore\1[^>]*>([\s\S]*?)<\/script>\s*/gi;
 11 	var tmpl=function(template,data){
 12 		if(template){
 13 			var resultTemplate;
 14 			resultTemplate=tmplCaches[template];
 15 			if(!resultTemplate){
 16 				resultTemplate=stack + ".push('" + template
 17 				.replace(/\s+/g," ")
 18 				.replace(/<#/g,"\r")
 19 				.replace(/;*#>/g,"\n")
 20 				.replace(/\\(?=[^\r\n]*\n)/g,"\t")
 21 				.replace(/\\/g,"\\\\")
 22 				.replace(/\t/g,"\\")
 23 				.replace(/'(?=[^\r\n]*\n)/g,"\t")
 24 				.replace(/'/g,"\\'")
 25 				.replace(/\t/g,"'")
 26 				.replace(/\r=([^\n]+)\n/g,"',$1,'")
 27 				.replace(/\r/g,"');")
 28 				.replace(/\n/g,";"+stack+".push('")+ "');return "+stack+".join('')";
 29 				 tmplCaches[template]=resultTemplate;
 30 			}
 31 			var vars=[stack],values=[[]],fnKey;
 32 			if(data){
 33 				for(var p in data){
 34 					vars.push(p.replace(/[:+\-*\/&^%#@!~]/g,'$'));
 35 					values.push(data[p]);
 36 				}
 37 			}
 38 			fnKey=vars.join('_')+'_'+resultTemplate;
 39 			if(!fnCaches[fnKey]){
 40 				try{
 41 					fnCaches[fnKey]=new Function(vars,resultTemplate);
 42 				}catch(e){
 43 					
 44 					return resultTemplate=e.message;
 45 				}
 46 			}
 47 			try{
 48 				resultTemplate=fnCaches[fnKey].apply(data,values);
 49 			}catch(e){
 50 				
 51 				resultTemplate=e.message;
 52 			}
 53 			return resultTemplate;
 54 		}
 55 		return template;
 56 	};
 57 	/**
 58 	 * 语法为<# #>的模板，<# #>语句 <#= #>输出
 59 	 * @name Tmpl
 60 	 * @namespace
 61 	 * @example
 62 	 * <#for(var i=0;i<10;i++){#>
 63 	 *    <#=i#> <br />
 64 	 * <#}#>
 65 	 */
 66 	var Tmpl={
 67 		/**
 68 		 * @lends Tmpl
 69 		 */
 70 		/**
 71 		 * 把模板与数据翻译成最终的字符串
 72 		 * @param {String} template 模板字符串
 73 		 * @param {Object} data     数据对象
 74 		 * @return {String}
 75 		 */
 76 		toHTML:function(template,data){
 77 			var notRenders=template.match(notRender);
 78 			if(notRenders){
 79 				template=template.replace(notRender,function(){//防止不必要的解析
 80 					return '<script type="text/ignore"></script>';
 81 				});
 82 				template=tmpl(template,data);
 83 				var idx=0;
 84 				template=template.replace(notRender,function(){
 85 					return notRenders[idx++];
 86 				});
 87 			}else{
 88 				template=tmpl(template,data);
 89 			}
 90 			return template;
 91 		}
 92 	};
 93 	return Tmpl;
 94 });
